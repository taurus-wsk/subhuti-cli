# templates/daily_build.yaml

name: daily_build
version: 1.0.0
description: "每日构建流水线：拉取代码 → 构建 → 测试 → 部署"
author: "Your Name"

# 全局配置

global:
timeout: 3600 # 全局超时时间（秒）
max_retries: 3 # 全局最大重试次数
on_failure: "notify_and_stop"  # 失败策略：notify_and_stop / continue / rollback
variables:  # 全局变量
project_root: "./my-project"
build_dir: "./build"
output_dir: "./dist"

# 定时触发配置（可选）

schedule:
enabled: true
cron: "0 2 * * *"  # 每天凌晨2点执行
timezone: "Asia/Shanghai"

# 事件触发配置（可选）

triggers:

- type: "file_change"
  path: "./src"
  events: ["create", "modify"]

# 人工干预钩子配置（可选）

hooks:
before_all: "confirm_start"  # 执行前确认
after_failure: "notify_failure"  # 失败后通知
after_success: "notify_success"  # 成功后通知

# Stage 定义（核心）

stages:

- id: "stage_init"
  name: "初始化阶段"
  description: "环境检查与初始化"
  on_failure: "stop"  # 本阶段失败策略
  steps:
    - id: "step_check_env"
      name: "检查环境"
      description: "检查 Go 环境是否可用"
      skill: "dev_tools.check_command"
      params:
      command: "go"
      args: ["version"]
      on_failure: "stop"
      timeout: 60

    - id: "step_clean"
      name: "清理构建目录"
      description: "清理旧的构建文件"
      skill: "file_ops.delete_dir"
      params:
      path: "{{global.build_dir}}"
      on_failure: "continue"  # 清理失败可以继续

- id: "stage_build"
  name: "构建阶段"
  description: "代码构建与依赖安装"
  depends_on: ["stage_init"]  # 依赖上一个阶段
  steps:
    - id: "step_pull_code"
      name: "拉取代码"
      description: "从 Git 拉取最新代码"
      skill: "dev_tools.git_pull"
      params:
      repo: "{{global.project_root}}"
      branch: "main"
      timeout: 300
      retry_on_failure: true

    - id: "step_install_deps"
      name: "安装依赖"
      description: "安装 Go 依赖"
      skill: "dev_tools.go_mod_tidy"
      params:
      path: "{{global.project_root}}"
      timeout: 600
      retry_on_failure: true

    - id: "step_build"
      name: "构建项目"
      description: "构建可执行文件"
      skill: "dev_tools.go_build"
      params:
      path: "{{global.project_root}}"
      output: "{{global.build_dir}}/app"
      timeout: 600
      retry_on_failure: true
      before_hook: "confirm_build"  # 构建前确认

- id: "stage_test"
  name: "测试阶段"
  description: "运行测试"
  depends_on: ["stage_build"]
  steps:
    - id: "step_run_tests"
      name: "运行单元测试"
      description: "运行所有单元测试"
      skill: "dev_tools.go_test"
      params:
      path: "{{global.project_root}}"
      coverage: true
      timeout: 600
      on_failure: "stop"

- id: "stage_deploy"
  name: "部署阶段"
  description: "部署到测试环境"
  depends_on: ["stage_test"]
  condition: "{{global.deploy_enabled}} == true"  # 条件执行
  steps:
    - id: "step_deploy"
      name: "部署"
      description: "部署到测试服务器"
      skill: "dev_tools.deploy"
      params:
      source: "{{global.build_dir}}/app"
      target: "/opt/app/"
      timeout: 300

# 输出配置

outputs:

- name: "build_artifact"
  type: "file"
  path: "{{global.build_dir}}/app"
- name: "test_report"
  type: "file"
  path: "{{global.build_dir}}/coverage.html"

# templates/blender_render.yaml

name: blender_render
version: 1.0.0
description: "Blender 场景渲染流水线"

global:
variables:
model_path: "./models/scene.fbx"
output_dir: "./renders"
resolution: "1920x1080"
engine: "CYCLES"

schedule:
enabled: false

stages:

- id: "stage_init"
  name: "初始化"
  steps:
    - id: "step_check_blender"
      name: "检查 Blender"
      skill: "blender_3d.check_blender"
      on_failure: "stop"

    - id: "step_clean_output"
      name: "清理输出目录"
      skill: "file_ops.delete_dir"
      params:
      path: "{{global.output_dir}}"
      on_failure: "continue"

- id: "stage_import"
  name: "导入模型"
  depends_on: ["stage_init"]
  steps:
    - id: "step_import"
      name: "导入 FBX 模型"
      skill: "blender_3d.import_model"
      params:
      path: "{{global.model_path}}"
      timeout: 600

- id: "stage_setup"
  name: "场景设置"
  depends_on: ["stage_import"]
  steps:
    - id: "step_add_light"
      name: "添加灯光"
      skill: "blender_3d.add_light"
      params:
      type: "sun"
      location: [5, 5, 10]
      energy: 3

    - id: "step_setup_camera"
      name: "设置相机"
      skill: "blender_3d.setup_camera"
      params:
      location: [5, -5, 5]
      look_at: [0, 0, 0]

- id: "stage_render"
  name: "渲染"
  depends_on: ["stage_setup"]
  steps:
    - id: "step_render"
      name: "渲染场景"
      skill: "blender_3d.render"
      params:
      output_path: "{{global.output_dir}}/render.png"
      resolution: "{{global.resolution}}"
      engine: "{{global.engine}}"
      timeout: 3600
      retry_on_failure: true
      before_hook: "confirm_render"

-------------
流水模式使用

# 1. 执行指定模板

subhuti pipeline run --template daily_build.yaml

# 2. 执行模板并覆盖变量

subhuti pipeline run --template daily_build.yaml --var project_root=./other-project --var deploy_enabled=true

# 3. 列出所有可用模板

subhuti pipeline list

# 4. 查看模板详情

subhuti pipeline show --template daily_build.yaml

# 5. 从断点续跑

subhuti pipeline resume --run-id 20240520-001 --from-stage stage_build

# 6. 启用定时执行

subhuti pipeline schedule --template daily_build.yaml --enable

--------
任务模式 使用

# 1. 自然语言描述任务

subhuti task do "帮我创建一个目录 demo，然后在里面创建一个 hello.txt 文件，内容是 Hello World"

# 2. 任务模式 + 指定插件

subhuti task do "渲染这个 Blender 模型" --plugin blender_3d

# 3. 任务模式 + 交互式确认

subhuti task do "批量重命名图片" --interactive

# 4. 查看历史任务

subhuti task history

# 5. 重新执行历史任务

subhuti task rerun --task-id 20240520-001
